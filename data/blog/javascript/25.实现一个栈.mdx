```javascript
// 实现栈，有入栈出栈的方法，以及length属性

class Stack {
  constructor() {
    this.items = []
  }
  push(item) {
    this.items.push(item)
  }
  pop() {
    return this.items.pop()
  }
  get length() {
    return this.items.length
  }
}

```
